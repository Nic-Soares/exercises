Sendo a diferena do (Multiple) R-squared obtido no exerccio anterior e nesta agora menor que 5%, este agora um modelo melhor?

# Calculando a diferena percentual dos R-squared
old_r_squared <- summary(fit)$r.squared
new_r_squared <- summary(fit_new)$r.squared

difference_percent <- ((new_r_squared - old_r_squared) / old_r_squared) * 100

# Verificando se a diferena  menor que 5%
if (abs(difference_percent) < 5) {
  print("Sim, o novo modelo  melhor.")
} else {
  print("No, o modelo anterior  melhor.")
}
